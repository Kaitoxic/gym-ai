        id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
        slug          TEXT UNIQUE NOT NULL,
        name          TEXT NOT NULL,
        muscle_groups TEXT[] NOT NULL,
        equipment     TEXT[] NOT NULL,
        difficulty    TEXT NOT NULL CHECK (difficulty IN (beginner, intermediate, advanced)),
        instructions  TEXT[] NOT NULL,
        image_url     TEXT,
        video_url     TEXT,
        created_at    TIMESTAMPTZ NOT NULL DEFAULT now()
      );

      ALTER TABLE exercises ENABLE ROW LEVEL SECURITY;

  <name>Task 3: GET /exercises endpoint + mount in index.ts</name>
  <files>src/routes/exercises.ts, src/index.ts</files>
  <action>
    src/routes/exercises.ts:
    - Import Router from express, supabase from ../lib/supabase
    - const router = Router()

    GET / (list endpoint):
    - Query params: muscle_group (string), equipment (string), difficulty (string), search (string), page (number, default 1), limit (number, default 20, max 100)
    - Base query: supabase.from(exercises).select(*, { count: exact })
    - Apply filters:
        if muscle_group: .contains(muscle_groups, [muscle_group])
        if equipment: .contains(equipment, [equipment])
        if difficulty: .eq(difficulty, difficulty)
        if search: .ilike(name, %search%)
    - Pagination: .range((page-1)*limit, page*limit-1)
    - Return: { data: Exercise[], total: count, page, limit }
    - No JWT required -- public endpoint

    GET /:slug (detail endpoint):
    - supabase.from(exercises).select(*).eq(slug, req.params.slug).single()
    - If not found: 404 { error: Exercise not found }
    - If found: 200 { data: Exercise }
    - No JWT required -- public endpoint

    Wrap both handlers in try/catch: 500 on unexpected error
    export default router

    src/index.ts update:
    - import exercisesRouter from ./routes/exercises
    - app.use(/exercises, exercisesRouter)  // after express.json()
